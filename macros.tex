% fancy diagram shortcuts
% double adjunction
\newcommand{\dadjnctn}[6][2em]{
  \xymatrix@1@C=#1{
    #2 \ar@<2ex>[r]^-{#4} \ar@<-2ex>[r]_-{#6}
    \ar@{}@<1.2ex>[r]|-{\scriptscriptstyle\bot} \ar@{}@<-1.2ex>[r]|-{\scriptscriptstyle\bot}
    &
    #3 \ar[l]|-{#5}
  }
}
